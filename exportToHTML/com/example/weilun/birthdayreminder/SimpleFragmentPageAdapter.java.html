54       @Override 
55       public Object instantiateItem(ViewGroup container, int position) { 
56           Fragment fragment = (Fragment) super.instantiateItem(container, position); 
57           registeredFragments.put(position, fragment); 
58           return fragment; 
59       } 
60    
61       @Override 
62       public void destroyItem(ViewGroup container, int position, Object object) { 
63           registeredFragments.remove(position); 
64           super.destroyItem(container, position, object); 
65       } 
66    
67       /** 
68        * get fragment from current tab 
69        * 
70        * @param position 
71        * @return current fragment 
72        */ 
73       public Fragment getRegisteredFragment(int position) { 
74           return registeredFragments.get(position); 
75       }
