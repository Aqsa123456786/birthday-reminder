16   public class DeleteRecordsDialogFragment extends DialogFragment { 
17    
18       @NonNull 
19       @Override 
20       public Dialog onCreateDialog(Bundle savedInstanceState) { 
21           AlertDialog.Builder builder = new AlertDialog.Builder(getActivity()); 
22           builder.setMessage(R.string.dialog_message).setTitle(R.string.dialog_title) 
23                   .setPositiveButton(R.string.btn_delete, new DialogInterface.OnClickListener() { 
24                       @Override 
25                       public void onClick(DialogInterface dialog, int which) { 
26                           notifyToTarget(Activity.RESULT_OK); 
27                       } 
28                   }) 
29                   .setNegativeButton(R.string.btn_cancel, new DialogInterface.OnClickListener() { 
30                       @Override 
31                       public void onClick(DialogInterface dialog, int which) { 
32                           notifyToTarget(Activity.RESULT_CANCELED); 
33                           dialog.dismiss(); 
34                       } 
35                   }); 
36           return builder.create(); 
37       } 
38    
39       /** 
40        * helper method to send result code to targeted fragment 
41        * 
42        * @param code 
43        */ 
44       private void notifyToTarget(int code) { 
45           Fragment targetFragment = getTargetFragment(); 
46           if (targetFragment != null) { 
47               targetFragment.onActivityResult(getTargetRequestCode(), code, null); 
48           } 
49       } 
50   }
