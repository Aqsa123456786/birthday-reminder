21   public class PersonCursorAdapter extends CursorAdapter { 
22       private LayoutInflater inflater; 
23    
24       public PersonCursorAdapter(Context context, Cursor cursor, int flags) { 
25           super(context, cursor, flags); 
26           inflater = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE); 
27       } 
28    
29       @Override 
30       public View newView(Context context, Cursor cursor, ViewGroup parent) { 
31           return inflater.inflate(R.layout.list_item, parent, false); 
32       } 
33    
34       @Override 
35       public void bindView(View view, Context context, Cursor cursor) { 
36           ImageView imageView = (ImageView) view.findViewById(R.id.image); 
37           TextView nameTv = (TextView) view.findViewById(R.id.name); 
38           TextView birthdayTv = (TextView) view.findViewById(R.id.birthday); 
39           TextView ageTv = (TextView) view.findViewById(R.id.age); 
40    
41           int imageResourceId = cursor.getInt(cursor.getColumnIndex(PersonContract.PersonEntry.COLUMN_NAME_IMAGERESOUCEID)); 
42           String name = cursor.getString(cursor.getColumnIndex(PersonContract.PersonEntry.COLUMN_NAME_NAME)); 
43           long dob = cursor.getLong(cursor.getColumnIndex(PersonContract.PersonEntry.COLUMN_NAME_DOB)); 
44    
45           imageView.setImageResource(imageResourceId); 
46           nameTv.setText(name); 
47           birthdayTv.setText(new SimpleDateFormat("EEEE, MMM,  d").format(dob)); 
48           ageTv.setText(Integer.toString(getAge(dob))); 
49       } 
50    
51       /** 
52        * helper method to get age from birhday 
53        * 
54        * @param dob 
55        * @return age 
56        */ 
57       private int getAge(long dob) { 
58           Calendar calendar = Calendar.getInstance(); 
59           Calendar age = Calendar.getInstance(); 
60           age.setTimeInMillis(dob); 
61           return calendar.get(Calendar.YEAR) - age.get(Calendar.YEAR); 
62       } 
63   }
