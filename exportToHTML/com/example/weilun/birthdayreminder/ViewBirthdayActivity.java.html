65           long idFromContactFragment = intent.getLongExtra(ContactListFragment.EXTRA_ID, 0);

71               id = idFromContactFragment; 
72               Log.v("VIewAcitivty", "id from contact fragment"); 
73           } else { 
74               id = idFromTodayBirthdayActivity; 
75               Log.v("VIewAcitivty", "id from TodayBirthdayActivity"); 
76           }

78           PersonDBQueries dbQueries = new PersonDBQueries(new PersonDBHelper(getApplicationContext())); 
79           String[] columns = PersonContract.columns;

83           Cursor cursor = dbQueries.query(columns, selection, selectionArgs, null, null, null);

150              } 
151          }.start();

155      /** 
156       * helper method to do countdown calculation 
157       * 
158       * @param millisUntilFinished 
159       */ 
160      private void onTickCalculation(long millisUntilFinished) { 
161          TextView tvDay = (TextView) findViewById(R.id.countdown_day); 
162          TextView tvHour = (TextView) findViewById(R.id.countdown_hour); 
163          TextView tvMinute = (TextView) findViewById(R.id.countdown_minute); 
164          TextView tvSecond = (TextView) findViewById(R.id.countdown_second); 
165   
166          long beginTime = System.currentTimeMillis(); 
167          beginTime = beginTime - 1; 
168          long serverUpTimeSeconds = (millisUntilFinished - beginTime) / 1000; 
169   
170          tvDay.setText(Long.toString(serverUpTimeSeconds / 86400)); 
171          tvHour.setText(Long.toString((serverUpTimeSeconds % 86400) / 3600)); 
172          tvMinute.setText(Long.toString(((serverUpTimeSeconds % 86400) % 3600) / 60)); 
173          tvSecond.setText(Long.toString(((serverUpTimeSeconds % 86400) % 3600) % 60)); 
174      }
