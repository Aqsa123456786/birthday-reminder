32       @Override 
33       protected void onCreate(Bundle savedInstanceState) { 
34           super.onCreate(savedInstanceState); 
35           setContentView(R.layout.activity_add_birthday); 
36           Toolbar toolbar = (Toolbar) findViewById(R.id.toolbar); 
37           setSupportActionBar(toolbar); 
38           sharedPreferences = getSharedPreferences("AddSavingState", Context.MODE_PRIVATE); 
39           editor = sharedPreferences.edit(); 
40    
41           image = (ImageView) findViewById(R.id.icon); 
42           etName = (EditText) findViewById(R.id.add_birthday_name); 
43           etEmail = (EditText) findViewById(R.id.add_birthday_email); 
44           etPhone = (EditText) findViewById(R.id.add_birthday_phone); 
45           etDob = (EditText) findViewById(R.id.birthday_date); 
46           aSwitch = (Switch) findViewById(R.id.show_noti); 
47    
48           FloatingActionButton fab = (FloatingActionButton) findViewById(R.id.fab); 
49           fab.setOnClickListener(new View.OnClickListener() { 
50               @Override 
51               public void onClick(View view) { 
52                   try { 
53                       //TODO ; set iamge reources id 
54                       int imageResourceId = R.drawable.ic_account_circle_black_24dp; 
55                       String name = etName.getText().toString(); 
56                       String email = etEmail.getText().toString(); 
57                       String phone = etPhone.getText().toString(); 
58                       SimpleDateFormat dateFormat = new SimpleDateFormat("EEEE, MMMM d, yyyy", Locale.ENGLISH); 
59                       Date date = dateFormat.parse(etDob.getText().toString()); 
60                       Boolean isChecked = aSwitch.isChecked(); 
61    
62                       PersonDBQueries dbQueries = new PersonDBQueries(new PersonDBHelper(getApplicationContext())); 
63                       Person person = new Person(name, email, phone, date, isChecked, imageResourceId); 
64                       if (dbQueries.insert(person) != 0) { 
65                           saved = true; 
66                           Toast.makeText(AddBirthdayActivity.this, "inserted", Toast.LENGTH_SHORT).show(); 
67                           finish(); 
68                       } 
69                   } catch (ParseException e) { 
70                       Toast.makeText(AddBirthdayActivity.this, e.getMessage(), Toast.LENGTH_SHORT).show(); 
71                   } 
72               } 
73           }); 
74       }

76       @Override 
77       protected void onPause() { 
78           super.onPause(); 
79           //TODO : implement full shared preference 
80           if(saved){ 
81               editor.clear(); 
82           } 
83           else { 
84               String name = etName.getText().toString(); 
85               String email = etEmail.getText().toString(); 
86               String phone = etPhone.getText().toString(); 
87    
88               editor.putString("SAVE_STATE_NAME", name); 
89               editor.putString("SAVE_STATE_EMAIL", email); 
90               editor.putString("SAVE_STATE_PHONE", phone); 
91           } 
92           editor.commit(); 
93       } 
94

98    
99           String name = sharedPreferences.getString("SAVE_STATE_NAME", ""); 
100          String email = sharedPreferences.getString("SAVE_STATE_EMAIL", ""); 
101          String phone = sharedPreferences.getString("SAVE_STATE_PHONE", ""); 
102   
103          etName.setText(name); 
104          etEmail.setText(email); 
105          etPhone.setText(phone); 
106      }
