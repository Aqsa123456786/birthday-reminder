3    import android.app.DatePickerDialog; 
4    import android.app.Dialog; 
5    import android.os.Bundle; 
6    import android.widget.DatePicker;

9    import java.text.SimpleDateFormat; 
10   import java.util.Calendar; 
11    
12   /** 
13    * Created by Wei Lun on 8/7/2017.

16   public class DatePickerFragment extends android.support.v4.app.DialogFragment 
17           implements DatePickerDialog.OnDateSetListener { 
18   //TODO: set max calendar year to current year 
19       private static Calendar calendar = null; 
20    
21       @Override 
22       public Dialog onCreateDialog(Bundle savedInstanceState) { 
23           if (calendar == null) 
24               calendar = Calendar.getInstance(); 
25           int year = calendar.get(Calendar.YEAR); 
26           int month = calendar.get(Calendar.MONTH); 
27           int day = calendar.get(Calendar.DAY_OF_MONTH); 
28    
29           return new DatePickerDialog(getActivity(), this, year, month, day); 
30       } 
31    
32       @Override 
33       public void onDateSet(DatePicker view, int year, int month, int dayOfMonth) { 
34           calendar.set(year, month, dayOfMonth); 
35           SimpleDateFormat dateFormat = new SimpleDateFormat("EEEE, MMMM d, yyyy"); 
36           String date = dateFormat.format(calendar.getTime()); 
37           EditText etDate = (EditText) getActivity().findViewById((R.id.birthday_date)); 
38           etDate.setText(date); 
39       } 
40   }
