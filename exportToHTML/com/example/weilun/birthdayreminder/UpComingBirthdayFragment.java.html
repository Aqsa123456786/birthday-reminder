36       @Override 
37       public void onAttach(Context context) { 
38           super.onAttach(context); 
39           try { 
40               countable = (Countable) context; 
41           } catch (ClassCastException e) { 
42               throw new ClassCastException("must implement Countable interface"); 
43           } 
44       }

46       @Override 
47       public View onCreateView(final LayoutInflater inflater, ViewGroup container, 
48                                Bundle savedInstanceState) { 
49           View rootView = inflater.inflate(R.layout.fragment_up_coming, container, false); 
50           ListView listView = (ListView) rootView.findViewById(R.id.listview); 
51           ProgressBar loadingBar = (ProgressBar) rootView.findViewById(R.id.loading_bar); 
52           loadingBar.setVisibility(View.GONE); 
53    
54           tv = (TextView) rootView.findViewById(R.id.no_birthday); 
55           listView.setEmptyView(tv); 
56           adapter = new PersonCursorAdapter(getActivity(), null, 0); 
57           tv.setText(getString(R.string.no_birthday_for_upcoming)); 
58           listView.setAdapter(adapter); 
59           listView.setOnItemClickListener(new AdapterView.OnItemClickListener() { 
60               @Override 
61               public void onItemClick(AdapterView<?> parent, View view, int position, long id) { 
62                   Cursor c = (Cursor) parent.getItemAtPosition(position); 
63                   Intent intent = new Intent(getActivity(), ViewBirthdayActivity.class); 
64                   intent.putExtra(EXTRA_ID, c.getLong(c.getColumnIndex(PersonContract.PersonEntry._ID))); 
65                   if (intent.resolveActivity(getActivity().getPackageManager()) != null) 
66                       startActivity(intent); 
67               } 
68           }); 
69           setHasOptionsMenu(true); 
70           return rootView; 
71       }

73       @Override 
74       public void onResume() { 
75           super.onResume(); 
76           onRefreshView(); 
77       } 
78    
79       /** 
80        * refresh current view 
81        */ 
82       public void onRefreshView() { 
83           startDate = Calendar.getInstance(); 
84           startDate.add(Calendar.DAY_OF_MONTH, -1); 
85           endDate = Calendar.getInstance(); 
86           endDate.add(Calendar.DAY_OF_MONTH, 1); 
87    
88           PersonDBQueries dbQuery = new PersonDBQueries(new PersonDBHelper(getActivity())); 
89           String[] columns = PersonContract.columns; 
90           String[] selectionArgs = {startDate.getTimeInMillis() + "", "" + endDate.getTimeInMillis()}; 
91    
92           //to convert millisecond to Unix timestamp, divide by 1000 
93           Cursor cursor = dbQuery.query(columns, "strftime('%m-%d'," + PersonContract.PersonEntry.COLUMN_NAME_DOB + "/1000, 'unixepoch')" 
94                           + " BETWEEN strftime('%m-%d',?/1000, 'unixepoch') AND strftime('%m-%d',?/1000, 'unixepoch')", 
95                   selectionArgs, null, null, null); 
96    
97           countable.getCount(cursor.getCount()); 
98           adapter.swapCursor(cursor); 
99           adapter.notifyDataSetChanged(); 
100      }
