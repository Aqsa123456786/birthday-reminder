9    import android.support.annotation.Nullable; 
10   import android.support.v4.app.NotificationManagerCompat; 
11   import android.util.Log; 
12    
13   import com.example.weilun.birthdayreminder.db.PersonDBHelper; 
14   import com.example.weilun.birthdayreminder.db.PersonDBQueries; 
15    
16   import java.util.Calendar;

22   public class NotifyIntentService extends IntentService { 
23    
24       public static final int NOTIFICATION_ID = 1; 
25       private static String LOG_TAG = NotifyIntentService.class.getSimpleName(); 
26       private static Calendar calender; 
27    
28       public NotifyIntentService() { 
29           super(NotifyIntentService.class.getName()); 
30       } 
31    
32       @Override 
33       protected void onHandleIntent(@Nullable Intent intent) { 
34           calender = Calendar.getInstance(); 
35           calender.add(Calendar.DAY_OF_MONTH, -1); 
36    
37           PersonDBQueries dbQuery = new PersonDBQueries(new PersonDBHelper(this)); 
38           Cursor cursor = dbQuery.queryTodayBirthday(calender); 
39    
40           int todayBirthday = cursor.getCount(); 
41    
42           Log.v(LOG_TAG, "today birthday: " + todayBirthday); 
43    
44           if (todayBirthday > 0) { 
45    
46               Notification.Builder builder = new Notification.Builder(this) 
47                       .setContentTitle(getString(R.string.notification_title)) 
48                       .setContentText(String.format(getString(R.string.notification_content), todayBirthday)) 
49                       .setAutoCancel(true) 
50                       .setSmallIcon(R.drawable.ic_announcement_black_24dp) 
51                       .setColor(getResources().getColor(R.color.colorPrimary)) 
52                       .setLargeIcon(BitmapFactory.decodeResource(getResources(), 
53                               R.mipmap.birthday_icon_launcher)) 
54                       .setDefaults(Notification.DEFAULT_SOUND | Notification.DEFAULT_VIBRATE); 
55    
56               Log.v(LOG_TAG, "notification built"); 
57    
58               Intent todayBirthdayIntent = new Intent(this, TodayBirthdayActivity.class); 
59               PendingIntent pendingIntent = PendingIntent.getActivity(this, 0, todayBirthdayIntent, PendingIntent.FLAG_UPDATE_CURRENT); 
60    
61               builder.setContentIntent(pendingIntent); 
62               Notification notification = builder.build(); 
63               NotificationManagerCompat managerCompat = NotificationManagerCompat.from(this); 
64               managerCompat.notify(NOTIFICATION_ID, notification); 
65    
66               builder.build().flags |= Notification.FLAG_AUTO_CANCEL; 
67           } 
68       } 
69   }
