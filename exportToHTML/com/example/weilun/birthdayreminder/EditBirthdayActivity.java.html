23   public class EditBirthdayActivity extends AppCompatActivity { 
24       private Person person; 
25       private ImageView image; 
26       private EditText etName, etEmail, etPhone, etDob; 
27       private Switch aSwitch; 
28    
29       @Override 
30       protected void onCreate(Bundle savedInstanceState) { 
31           super.onCreate(savedInstanceState); 
32           setContentView(R.layout.activity_edit_birthday); 
33           Toolbar toolbar = (Toolbar) findViewById(R.id.toolbar); 
34           setSupportActionBar(toolbar); 
35           Intent intent = getIntent(); 
36           person = (Person) intent.getSerializableExtra(ViewBirthdayActivity.EXTRA_ID); 
37           setView(); 
38           getSupportActionBar().setDisplayHomeAsUpEnabled(true); 
39    
40           FloatingActionButton fab = (FloatingActionButton) findViewById(R.id.fab); 
41           fab.setOnClickListener(new View.OnClickListener() { 
42               @Override 
43               public void onClick(View view) { 
44                   try { 
45    
46                       //TODO: set image resources id 
47                       person.setImageResourceId(R.drawable.ic_account_circle_black_24dp); 
48                       person.setName(etName.getText().toString()); 
49                       person.setEmail(etEmail.getText().toString()); 
50                       person.setPhone(etPhone.getText().toString()); 
51                       person.setDob(new SimpleDateFormat("EEEE, MMMM d, yyyy", Locale.ENGLISH).parse(etDob.getText().toString())); 
52                       person.setNotify(aSwitch.isChecked()); 
53    
54                       PersonDBQueries dbQueries = new PersonDBQueries(new PersonDBHelper(getApplicationContext())); 
55                       if (dbQueries.update(person) != 0) { 
56                           Toast.makeText(EditBirthdayActivity.this, "Updated", Toast.LENGTH_SHORT).show(); 
57                           finish(); 
58                       } else 
59                           Toast.makeText(EditBirthdayActivity.this, "error", Toast.LENGTH_SHORT).show(); 
60                   } catch (ParseException e) { 
61                       Log.v("EditBIrthdayActivity", e.getMessage()); 
62                   } 
63               } 
64           }); 
65       }

72       /** 
73        * helper mehtod to setup UI

75       private void setView() { 
76           image = (ImageView) findViewById(R.id.icon); 
77           etName = (EditText) findViewById(R.id.add_birthday_name); 
78           etEmail = (EditText) findViewById(R.id.add_birthday_email); 
79           etPhone = (EditText) findViewById(R.id.add_birthday_phone); 
80           etDob = (EditText) findViewById(R.id.birthday_date); 
81           aSwitch = (Switch) findViewById(R.id.show_noti); 
82    
83           image.setImageResource(person.getImageResourceId()); 
84           etName.setText(person.getName()); 
85           etEmail.setText(person.getEmail()); 
86           etPhone.setText(person.getPhone()); 
87           etDob.setText(new SimpleDateFormat("EEEE, MMMM d, yyyy").format(person.getDOB())); 
88           aSwitch.setChecked(person.isNotify()); 
89       }
