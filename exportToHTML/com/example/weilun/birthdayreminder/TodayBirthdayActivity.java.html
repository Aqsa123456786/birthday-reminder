42               @Override

44                   Cursor c = (Cursor) parent.getItemAtPosition(position); 
45                   Intent intent = new Intent(TodayBirthdayActivity.this, ViewBirthdayActivity.class); 
46                   intent.putExtra(EXTRA_ID, c.getLong(c.getColumnIndex(PersonContract.PersonEntry._ID))); 
47                   if (intent.resolveActivity(getPackageManager()) != null) 
48                       startActivity(intent); 
49               }
